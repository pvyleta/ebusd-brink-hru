    {
      bool flag = true;
      if (this._slaveNumber != (short) 0)
        flag = BCSDialog.ShowQuestion("ChangeSlaveNumberDialogTitle", "ChangeSlaveNumberDialogWarningQuestion");
      if (flag)
      {
        this._wtwDataItemMng.SendRequestCommand(WTWDataItem.DataItemType.WriteParameter, new byte[3]
        {
          this._eBusSlaveNumberIndex,
          (byte) ((uint) this._slaveNumber >> 8),
          (byte) ((uint) this._slaveNumber & (uint) byte.MaxValue)
        });
        this.DialogResult = new bool?(true);
      }
      else
        this.DialogResult = new bool?(false);
    }
